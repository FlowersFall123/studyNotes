# 导航栏

## 核心功能

1. **动态导航结构**：
   - 支持一级和二级菜单
   - 通过 `rootNav` ref 定义导航结构
   - 自动根据当前路由展开对应父级菜单
2. **路由集成**：
   - 使用 Vue Router 的 `useRouter` 和 `useRoute`
   - 自动高亮当前路由对应的菜单项
3. **图标系统**：
   - 内联 SVG 图标，通过名称映射
   - 统一 24x24 视口，使用 currentColor
4. **响应式设计**：
   - 默认折叠状态 (80px 宽度)
   - 悬停时展开 (200px 宽度)
   - 平滑过渡动画

const currentParent = ref(null)
const activeList = computed(() => (currentParent.value?.children ?? filteredRootNav.value))

function isActive(item) {
  return item.route ? (route.path === item.route || route.path.startsWith(item.route + '/')) : false
}

function onItemClick(item) {
  if (item.children && item.children.length) {
    currentParent.value = item
  } else if (item.action === 'toggleTheme') {
    themeStore.toggleTheme()
  } else if (item.route) {
    router.push(item.route)
  }
}

function goBack() {
  currentParent.value = null
}

// 根据当前路由自动展开对应的父级
watch(
  () => route.path,
  (path) => {
    const parent = filteredRootNav.value.find((p) => (p.children ?? []).some((c) => path.startsWith(c.route)))
    currentParent.value = parent ?? null
  },
  {immediate: true}
)
